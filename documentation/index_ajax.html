H) End of Page (Javascript) - top

Javascript files loaded in the end of page. This will reduce page load time. 

  <!-- ================== BEGIN BASE JS ================== -->
  <script src="../assets/js/app.min.js"></script>
  <script src="../assets/js/theme/default.min.js"></script>
  <script>
    App.settings({
      ajaxMode: true,
      ajaxDefaultUrl: '#pages/index.html',
      ajaxType: 'GET',
      ajaxDataType: 'html'
    });
  </script>
  <!-- ================== END BASE JS ================== -->
</body>
</html>

I) Page Options - top

Here is the general javascript option for the each case of page options.

Reset to default after page leave

<script>
  App.setPageOption({
    clearOptionOnLeave: true
  });
</script>

Sidebar Light

<script>
  App.setPageOption({
    pageSidebarLight: true
  });
</script>

Sidebar Transparent

<script>
  App.setPageOption({
    pageSidebarTransparent: true
  });
</script>

Sidebar Minified

<script>
  App.setPageOption({
    pageSidebarMinified: true
  });
</script>

Sidebar Position Right

<script>
  App.setPageOption({
   pageSidebarRight: true
  });
</script>

Sidebar Wide

<script>
  App.setPageOption({
   pageSidebarWide: true
  });
</script>

Boxed Layout

<script>
  App.setPageOption({
   pageBoxedLayout: true
  });
</script>

Full Height Page / Table Column Page NEW

<script>
  App.setPageOption({
    pageContentFullWidth: true,
    pageContentFullHeight: true,
  });
</script>
